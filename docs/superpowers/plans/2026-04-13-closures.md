# Closure / Nested Function Contracts — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Inject `@pre`/`@post` contracts on nested function-like nodes (named inner `FunctionDeclaration`, `const`-assigned arrow/function expression, and returned arrow function) that appear inside an outer exported function or public method body.

**Architecture:** Two-phase rewrite. Phase 1 (existing) rewrites the outer function's own contracts as today. Phase 2 (new) is a single-level walk of the resulting body, looking for tagged nested function-like nodes and rewriting each in place. Phase 2 lives entirely in `src/function-rewriter.ts` as `rewriteNestedFunctions` (body walker) and `rewriteNestedFunctionLike` (per-node rewriter). Two new helpers are added to `src/node-helpers.ts`: `buildNestedLocationName` (produces `OuterName > innerName` strings) and `buildCapturedIdentifiers` (collects outer params and preceding variable bindings into the known-identifier set). The `#13` misuse-detection conditional in `src/transformer.ts` is narrowed so it no longer fires for the three supported nested forms. No other files require changes.

> **Note:** This plan should be implemented **after issue #14 (arrow functions)**, which introduces `normaliseArrowBody`, the `ArrowFunction`/`FunctionExpression` branches in `applyNewBody`, and `extractContractTagsForFunctionLike`. This plan reuses all three without modification.

```typescript
export function buildCapturedIdentifiers(
  outerNode: typescript.FunctionLikeDeclaration,
  innerStatementIndex: number,
): Set<string> {
  const captured = new Set<string>();

  // 1. Outer function parameters
  for (const param of outerNode.parameters) {
    extractBindingNames(param.name, captured);
  }

  // 2. Variable declarations that precede the inner function in the outer body
  const outerBody = outerNode.body;
  if (outerBody === undefined || !typescript.isBlock(outerBody)) {
    return captured;
  }
  for (let idx = 0; idx < innerStatementIndex; idx++) {
    const stmt = outerBody.statements[idx];
    if (typescript.isVariableStatement(stmt)) {
      for (const decl of stmt.declarationList.declarations) {
        extractBindingNames(decl.name, captured);
      }
    }
  }
  return captured;
}
```

```typescript
export function buildNestedLocationName(
  outerNode: typescript.FunctionLikeDeclaration,
  innerNode: typescript.FunctionLikeDeclaration,
  variableName?: string,
): string {
  const outerName = buildLocationName(outerNode);

  let innerName: string;
  if (
    typescript.isFunctionDeclaration(innerNode) &&
    innerNode.name !== undefined &&
    typescript.isIdentifier(innerNode.name)
  ) {
    innerName = innerNode.name.text;
  } else if (variableName !== undefined) {
    innerName = variableName;
  } else {
    innerName = '(anonymous)';
  }

  return `${outerName} > ${innerName}`;
}
```

```typescript
function rewriteNestedFunctionLike(
  factory: typescript.NodeFactory,
  node: typescript.FunctionLikeDeclaration,
  outerNode: typescript.FunctionLikeDeclaration,
  outerName: string,
  reparsedFunctions: Map<number, typescript.FunctionLikeDeclaration>,
  warn: (msg: string) => void,
  checker: typescript.TypeChecker | undefined,
  capturedIdentifiers: Set<string>,
  allowIdentifiers: string[],
  variableName?: string,
): typescript.FunctionLikeDeclaration | null {
  const originalBody = node.body;
  if (!originalBody || !typescript.isBlock(originalBody)) {
    return null;
  }

  const reparsedNode = reparsedFunctions.get(node.pos) ?? node;
  const tags = extractContractTagsForFunctionLike(reparsedNode);
  if (tags.length === 0) {
    return null;
  }

  const location = buildNestedLocationName(outerNode, node, variableName);
  const preKnown = buildKnownIdentifiers(node, false);
  const postKnown = buildKnownIdentifiers(node, true);

  for (const captured of capturedIdentifiers) {
    preKnown.add(captured);
    postKnown.add(captured);
  }

  const exportedNames = mergeIdentifiers(preKnown, postKnown, checker, node, allowIdentifiers);
  const paramTypes = checker !== undefined ? buildParameterTypes(node, checker) : undefined;
  const postParamTypes = buildPostParamTypes(node, checker, paramTypes);

  const preTags = filterValidTags(
    tags.filter((tag) => tag.kind === KIND_PRE),
    KIND_PRE, location, warn, preKnown, paramTypes, checker, node,
  );
  const postTagsWithResult = filterPostTagsWithResult(
    tags.filter((tag) => tag.kind === KIND_POST), node, location, warn,
  );

  const anyUsesPrev = postTagsWithResult.some((tag) => expressionUsesPrev(tag.expression));
  let prevCapture: string | null = null;
  if (anyUsesPrev) {
    prevCapture = resolvePrevCapture(node, reparsedNode, undefined, location, warn);
  }
  const postTags = filterValidTags(
    filterPostTagsRequiringPrev(postTagsWithResult, prevCapture, location, warn),
    KIND_POST, location, warn, postKnown, postParamTypes, checker, node,
  );

  if (shouldSkipRewrite(preTags, postTags, null)) {
    return null;
  }

  const newStatements = buildGuardedStatements(
    factory, preTags, postTags, originalBody, location, null, prevCapture, exportedNames,
  );
  return applyNewBody(factory, node, factory.createBlock(newStatements, true));
}
```

```typescript
function rewriteNestedFunctions(
  factory: typescript.NodeFactory,
  outerNode: typescript.FunctionLikeDeclaration,
  body: typescript.Block,
  reparsedFunctions: Map<number, typescript.FunctionLikeDeclaration>,
  transformed: { value: boolean },
  warn: (msg: string) => void,
  checker: typescript.TypeChecker | undefined,
  allowIdentifiers: string[],
): typescript.Block {
  let changed = false;
  const newStatements: typescript.Statement[] = [];

  body.statements.forEach((stmt, idx) => {
    const captured = buildCapturedIdentifiers(outerNode, idx);

    // Rule A: named FunctionDeclaration directly in the body
    if (typescript.isFunctionDeclaration(stmt) && stmt.body !== undefined) {
      const rewritten = rewriteNestedFunctionLike(
        factory, stmt, outerNode, buildLocationName(outerNode),
        reparsedFunctions, warn, checker, captured, allowIdentifiers,
      ) as typescript.FunctionDeclaration | null;
      if (rewritten !== null) {
        changed = true;
        transformed.value = true;
        newStatements.push(rewritten);
        return;
      }
      newStatements.push(stmt);
      return;
    }

    // Rule B: VariableStatement containing an arrow/function-expression initialiser
    if (typescript.isVariableStatement(stmt)) {
      const decls = stmt.declarationList.declarations;
      let declChanged = false;
      const newDecls = decls.map((decl) => {
        const init = decl.initializer;
        if (init === undefined) {
          return decl;
        }
        const isArrowOrFn =
          typescript.isArrowFunction(init) || typescript.isFunctionExpression(init);
        if (!isArrowOrFn) {
          return decl;
        }
        const varName =
          typescript.isIdentifier(decl.name) ? decl.name.text : undefined;
        const normNode = typescript.isArrowFunction(init)
          ? normaliseArrowBody(factory, init)
          : init;
        const rewritten = rewriteNestedFunctionLike(
          factory, normNode as typescript.FunctionLikeDeclaration,
          outerNode, buildLocationName(outerNode),
          reparsedFunctions, warn, checker, captured, allowIdentifiers, varName,
        );
        if (rewritten === null) {
          return decl;
        }
        declChanged = true;
        return factory.updateVariableDeclaration(
          decl, decl.name, decl.exclamationToken, decl.type,
          rewritten as typescript.Expression,
        );
      });
      if (declChanged) {
        changed = true;
        transformed.value = true;
        newStatements.push(
          factory.updateVariableStatement(
            stmt,
            typescript.getModifiers(stmt),
            factory.updateVariableDeclarationList(stmt.declarationList, newDecls),
          ),
        );
        return;
      }
      newStatements.push(stmt);
      return;
    }

    // Rule C: ReturnStatement containing an arrow/function-expression
    if (
      typescript.isReturnStatement(stmt) &&
      stmt.expression !== undefined &&
      (
        typescript.isArrowFunction(stmt.expression) ||
        typescript.isFunctionExpression(stmt.expression)
      )
    ) {
      const funcExpr = stmt.expression;
      const reparsedStmt =
        reparsedFunctions.get(stmt.getStart()) ??
        reparsedFunctions.get(funcExpr.pos) ??
        funcExpr;
      const normNode = typescript.isArrowFunction(funcExpr)
        ? normaliseArrowBody(factory, funcExpr)
        : funcExpr;
      const rewritten = rewriteNestedFunctionLike(
        factory, normNode as typescript.FunctionLikeDeclaration,
        outerNode, buildLocationName(outerNode),
        reparsedFunctions, warn, checker, captured, allowIdentifiers,
      );
      if (rewritten !== null) {
        changed = true;
        transformed.value = true;
        newStatements.push(
          factory.updateReturnStatement(stmt, rewritten as typescript.Expression),
        );
        return;
      }
    }

    newStatements.push(stmt);
  });

  if (!changed) {
    return body;
  }
  return factory.updateBlock(body, newStatements);
}
```

```typescript
export function tryRewriteFunction(
  factory: typescript.NodeFactory,
  node: typescript.FunctionLikeDeclaration,
  reparsedFunctions: Map<number, typescript.FunctionLikeDeclaration>,
  transformed: { value: boolean },
  warn: (msg: string) => void,
  checker?: typescript.TypeChecker,
  invariantExpressions: string[] = [],
  interfaceMethodContracts?: InterfaceMethodContracts,
  allowIdentifiers: string[] = [],
): typescript.FunctionLikeDeclaration {
  try {
    // Phase 1: rewrite outer function's own contracts
    const rewritten = rewriteFunction(
      factory, node, reparsedFunctions, warn, checker,
      invariantExpressions, interfaceMethodContracts, allowIdentifiers,
    );
    const workingNode = rewritten ?? node;
    if (rewritten !== null) {
      transformed.value = true;
    }

    // Phase 2: rewrite nested function-like nodes in the body
    const workingBody = workingNode.body;
    if (workingBody === undefined || !typescript.isBlock(workingBody)) {
      return workingNode;
    }
    const newBody = rewriteNestedFunctions(
      factory, workingNode, workingBody,
      reparsedFunctions, transformed, warn, checker, allowIdentifiers,
    );
    if (newBody === workingBody) {
      return workingNode;
    }
    const updated = applyNewBody(factory, workingNode, newBody);
    return updated ?? workingNode;
  } catch {
    return node;
  }
}
```
